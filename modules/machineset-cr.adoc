  replicas: 1
  selector:
    matchLabels:
      machine.openshift.io/cluster-api-cluster: cluster_name <1>
      machine.openshift.io/cluster-api-machineset: <cluster_name>-<machine_label>-<AWS-availability-zone> <3>
  template:
    metadata:
      creationTimestamp: null
      labels:
        machine.openshift.io/cluster-api-cluster: <cluster_name> <1>

        machine.openshift.io/cluster-api-machineset: <cluster_name>-<machine_label>-<AWS-availability-zone> <3>

      metadata:
        creationTimestamp: null
      providerSpec: <5>
        value:
          ami:
            id: ami-0e2bcd33dfff9c73e <6>

          metadata:
